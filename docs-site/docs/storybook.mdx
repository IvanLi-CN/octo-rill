title: Storybook
description: 跳转到 OctoRill Storybook 静态站。
---

import { useEffect } from "react";

export default function StorybookRedirect() {
  const storybookHref = "./storybook/";

  useEffect(() => {
    if (typeof window === "undefined") return;
    window.location.replace(storybookHref);
  }, [storybookHref]);

  return (
    <>
      <h1>正在跳转到 Storybook...</h1>
      <p>如果没有自动跳转，请点击下面的按钮手动进入 OctoRill Storybook。</p>
      <p>
        <button
          type="button"
          onClick={() => {
            if (typeof window === "undefined") return;
            window.location.assign(storybookHref);
          }}
        >
          打开 OctoRill Storybook
        </button>
      </p>
    </>
  );
}
