import { useEffect } from "react";

export const StorybookRedirect = () => {
  useEffect(() => {
    if (typeof window === "undefined") return;
    const target = new URL("./storybook/index.html", window.location.href);
    window.location.replace(target.toString());
  }, []);

  return (
    <>
      <h1>正在跳转到 Storybook…</h1>
      <p>如果没有自动跳转，请直接打开 <a href="./storybook/index.html">OctoRill Storybook</a>。</p>
      <p>如需查看文档导览页，请打开 <a href="./storybook-guide.html">Storybook 导览</a>。</p>
    </>
  );
};

<StorybookRedirect />
