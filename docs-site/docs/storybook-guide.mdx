title: Storybook 旧入口
description: 旧的 Storybook 导览链接已迁移到直接入口。
---

import { useEffect } from "react";

export const LegacyStorybookGuideRedirect = () => {
  useEffect(() => {
    if (typeof window === "undefined") return;
    window.location.replace("./storybook.html");
  }, []);

  return (
    <>
      <h1>旧入口已迁移</h1>
      <p>Storybook 导览页已经移除；如果没有自动跳转，请直接打开 <a href="./storybook.html">Storybook</a>。</p>
    </>
  );
};

<LegacyStorybookGuideRedirect />
